/**
 *  @namespace Util.cUtilDate
 *  @name Util.cUtilDate
 */
define([
  "common/c.class.inherit",
], function (
  inherit
) {

  var CData = new inherit.Class({
    /**
     * 生成一个CData实例
     * @class CData
     * @memberof Util.cUtilData
     * @param {Date} [date] 日期
     * @type {Core.Class}
     */
     initialize: function () {
     }
  });

  inherit.extend(CData,{
    //translate map to items
    mapToItems : function(map){
        var items = [];
        var keys = _.keys(map);
        var values = _.values(map);
        for(var i = keys.length - 1;i>=0;i--){
            items.push({id:keys[i],name:values[i]});
        }
        return items;
    }
  });
  return CData;
});
