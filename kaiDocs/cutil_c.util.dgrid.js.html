/**
 * 与验证相关的工具方法
 * @namespace Util.dgrid
 */
define([
  "custom/dstore/RestStore",
  "custom/dgrid/cdgrid",
  "dojo/_base/lang",
], function (
  RestStore,
  cdgrid,
  lang) {
    var util = {
            //格式返回数据
            format: function (res) {
              var self = this;
              var res = JSON.parse(res);
              if (res && res.responseStatus && res.responseStatus.code == '400004') {//TBD 暂时放在这里，后面要统一处理，该请求通过dojo发出
                self.showLoginDialog();
                return false;
              }
              if(_.isObject(res) && _.isString(res.responseStatus.msg)){
                Kai.showToast({
                  text:res.responseStatus.msg
                });
                return false;
              }
              var listCopy = $.extend(res.body.obj.list, {}, true);
              return listCopy;
            },

            createRestGrid:function(opts){
              /*
              {
              restModel://请求model,
              parseFn://原始请求的处理函数
              columns://definition of columns,
              rowsPerPage: 25, //每页显示多少条目
            }
              */
              var self = this;
              var RestCollectionStore = new (RestStore)(lang.mixin({
                target: opts.restModel
              }, {}));
              var restCollection = RestCollectionStore.filter({});
              restCollection.filter = opts.filter || {};
              if(opts.parseFn){
                restCollection.parse = opts.parseFn;
              }else{
                if(opts && _.isObject(opts)){
                  restCollection.parse = $.proxy(util.format,opts.view);
                }else{
                  restCollection.parse = util.format;
                }
              }
              //构造函数调用
              var restGrid = new (cdgrid)({
                collection: restCollection,
                columns: opts.columns,
                className: "dgrid-autoheight",
                rowsPerPage:opts.rowsPerPage || 25,
                pageSizeOptions:opts.pageSizeOptions || [5, 10, 15, 25], /*显示可以选择的分页*/
              }, opts.elementId);
              return restGrid;
            }
    };

    /**
    @description 更新表格
    @param grid
    @param opts object {list:data}
    */
    util.updateGrid = function(grid,opts){
      if(opts && opts.isRest){
        /*
        need opts to update rest grid
        */
        if(opts.filterParam){
          if(!grid.collection){
            return false;
          }
          grid.collection.filter = _.isObject(opts)?opts.filterParam:{};
        }
        grid && grid.refresh().then(function(){
          self.grid && self.grid.startup();
        },function(){
        });
      }else{
        grid && grid.collection && grid.collection.setData(opts.list);
        grid && grid.refresh();
        //After putting new data, you can just call on grid.refresh() and it should update your table.
      }
    };
    util.formatter = function(value) {
        return '<span title="' + _.escape(value) + '">' + _.escape(value) + '</span>';
    };
    return util;
});
