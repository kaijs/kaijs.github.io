/**
 * 影像扫描
 */
define([
  "module"
], function(
  module
) {
    var imageScanUtil = {};

    /** 打开影像系统
        role:影像角色
        flag:是否需要验证用户权限
        type:影像类型，scan为扫描，query为查看
    */
    imageScanUtil.imageScanQuery = function(applyCode, role, flag, type) {
      var self = this;
      var Models = module.config().models;
      var imageScanModel = Models.imageScan.getInstance();
        var onSuccess = function(resp) {
            if (resp && resp.responseStatus.code == "000000") {
                var result = resp.body.obj;

                if (flag) {
                    // 角色包括客服（id＝42383）并且节点处于申请录入或录入复核可以上传影像
                    if (result.userRole && (result.userRole.indexOf('42383') != -1) && result.approveNode && (result.approveNode === 'fillCustomerInfoTaskOfCL' || result.approveNode === 'fillCustomerInfoCheckTaskOfCL')) {
                        type = 'scan';
                    } else {
                        type = 'query';
                    }
                }

                // 跳转到影像系统
                $('form[id=imageFm]').remove(); //每次在页面存留一个form
                var codeVal;
                if (type == 'query') { //影像系统仅查询
                    codeVal = 'ECM0002';
                } else {
                    codeVal = 'ECM0001';
                }
                var imageFm = $('<form/>', {
                    method: 'POST',
                    target: '_blank',
                    id: 'imageFm',
                    action: result.url
                }).appendTo('body');

                var code = $('<input/>', {
                    name: 'code',
                    type: 'hidden',
                    value: codeVal
                }).appendTo(imageFm);
                var format = $('<input/>', {
                    name: 'format',
                    type: 'hidden',
                    value: 'xml'
                }).appendTo(imageFm);
                var xml = $('<input/>', {
                    name: 'xml',
                    type: 'hidden',
                    value: result.xml
                }).appendTo(imageFm);

                imageFm.submit();
            } else {
                self.showToast({
                    text: resp.body.obj.msg
                });
            }
        };
        var onError = function() {
            self.showToast({
                text: "请求服务器错误"
            });
        };
        imageScanModel.param = {
            "busino": applyCode,
            "rolecode": role
        };
        imageScanModel.execute(onSuccess, onError, true, self);
    };

    /*
    {
    applyCode:'',
    role:'',
    qsFlag:''
    }
    */
    imageScanUtil.openQueryWin = function(opts) {
        var applyCode = opts.applyCode;
        var Models = module.config().models;
        var getImageSystemAuthModel = Models.getImageSystemAuth.getInstance();
        if (applyCode == null || applyCode == '' || applyCode == undefined) {
            alert('检查不到订单编号，请联系管理员!');
            return false;
        }

        var onSuccess = function(result) {
            if (result.code == '000000') {
                imageScanUtil.imageScanQuery(result.data.imageurl, result.data.xml, opts.qsFlag);
            } else {
                alert('影像系统配置异常，请检查!');
            }
        };

        var onError = function() {
            alert('影像系统配置异常，请检查!');
        };

        getImageSystemAuthModel.param = {
            applyCode: opts.applyCode,
            role: opts.role,
            qsFlag: opts.qsFlag
        };

        getImageSystemAuthModel.execute(onSuccess, onError, true, this);
    }

    return imageScanUtil;


})
