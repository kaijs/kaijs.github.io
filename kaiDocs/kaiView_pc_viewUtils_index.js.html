    + ------------------------------------------------------- +
    1) 任何少于6个字符的组合，弱；任何字符数的同类字符组合，弱；
    2) 任何字符数的两类字符组合，中；
    3) 12位字符数以下的三类或四类字符组合，强；
    4) 12位字符数以上的三类或四类字符组合，非常好。
    + ------------------------------------------------------- +
    **/
    //检测密码强度
    checkStrong:function(sValue)
    {
      var modes = 0;
      if (sValue.length < 8) return modes;
      if (/\d/.test(sValue)) modes++; //数字
      if (/[a-z]/.test(sValue)) modes++; //小写
      if (/[A-Z]/.test(sValue)) modes++; //大写
      if (/[_\W]/.test(sValue)) modes++; //特殊字符
      switch (modes)
      {
        case 1:
        return 1;
        break;
        case 2:
        return 2;
        case 3:
        case 4:
        return sValue.length < 12 ? 3 : 4//12 --> 8
        break;
      }
    },

    //安全登出
    safeLogout:function(){
      var self = this;
      var UserStore = CommonStore.UserStore.getInstance();//获得用户信息存储
      UserStore.remove();//删除用户信息
      _.isObject(localStorage) && localStorage.clear();//清空所有本地缓存
      self.forward(Kai.loginUrl,{//跳转登录页面
        keepURL:true
      });
    },

    //打开所有tabs
    openAllTabs:function(){
      var self = this;
      var navigationItems = self.store.getAttr('navigationItems');
      _.each(navigationItems,function(navigationItem,index){
        if(navigationItem.leaf){
          _.delay(function(){
            self.model && self.model.set({
              'currentChildViewName':navigationItem.url//新选择的tab
            },
            {changeType:'tab'}
            );/*current view is changed*/
          },1000*(index + 1));
        }
      });
    }
  };

  return viewUtils;
