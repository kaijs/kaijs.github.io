         * 定义WeixinApi
         */
        var WeixinApi = {
            version: 6.2
        };

        /**
         * 对外暴露对微信接口的操作
         */
        WeixinApi.wx = jWeixin;

        /**
         * 判断是否在微信中
         */
        WeixinApi.isInWeixin = function () {
            var ua = navigator.userAgent.toLowerCase();
            if (ua.match(/MicroMessenger/i) == "micromessenger") {
                return true;
            } else {
                return false;
            }
        };

        /**
         * 获取后台签名及签名信息，通过config接口注入权限验证配置
         */
        WeixinApi.wxConfig = function (currentUrl) {

            var onSuccess = function (json) {
                if (json && json.Body) {
                    //alert(JSON.stringify(json.Body));
                    return WeixinApi.Config(json.Body);
                } else {
                    return false;
                }
            };

            var onError = function () {
                return false;
            };
            _weichatSignModel.param = { 'Url': currentUrl };
            _weichatSignModel.excute(onSuccess, onError, true, this);
        }

        /**
         * 通过config接口注入权限验证配置
         */
        WeixinApi.Config = function (jsonSignature) {
            //alert('Config' + 'timestamp' + jsonSignature.Timestamp + ';nonceStr' + jsonSignature.Noncestr + ';signature' + jsonSignature.Signature);
            if (jsonSignature && jsonSignature.Timestamp && jsonSignature.Noncestr && jsonSignature.Signature) {
                jWeixin.config({
                    //debug: true,
                    appId: 'wx34a8c5e7bdb33275',           // 必填，公众号的唯一标识
                    timestamp: jsonSignature.Timestamp,    // 必填，生成签名的时间戳
                    nonceStr: jsonSignature.Noncestr,      // 必填，生成签名的随机串
                    signature: jsonSignature.Signature,    // 必填，签名
                    jsApiList: [                           // 必填，需要使用的JS接口列表
                      'checkJsApi',
                      'onMenuShareTimeline',
                      'onMenuShareAppMessage',
                      'onMenuShareQQ',
                      'onMenuShareWeibo',
                      'hideMenuItems',
                      'showMenuItems',
                      'hideAllNonBaseMenuItem',

                      'translateVoice',
                      'startRecord',

                      'openLocation',
                      'getLocation',
                      'hideOptionMenu',

                      'closeWindow',
                      'scanQRCode',
                      'chooseWXPay',
                      'openProductSpecificView',
                      'addCard',
                      'chooseCard',
                      'openCard'

                });
                return true;
            } else {
                return false;
            }
