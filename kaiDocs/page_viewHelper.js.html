define([], function () {
    "use strict";
    var viewHelper = {};

    var listHelper = function (e, initialData) {
        this.self = e;
        if (_.isFunction(this.init)) {
            this.init(initialData);
        }
    }

    /**
     默认接口格式：
     请求：
     {
       pagenation :{
         startIndex:
         endIndex:
       }
     }
     返回结果:
     {
       body:{
         obj:{
           list:[]
           page:{
             startIndex:
             endIndex:
             dataCount
           }
         }
       }
     }
     */
    listHelper.prototype = {
        /*设置初始化参数*/
        init: function (initialData) {

            this.listContainer = initialData.listContainer;//当前列表容器
            this.listTplFun = initialData.listTplFun;//当前列表DOM生成函数
            this.NO_MORE_DOM = initialData.NO_MORE_DOM;//没有更多结果的显示DOM节点
            this.NO_RESULT_DOM = initialData.NO_RESULT_DOM;//当前列表为空时显示DOM节点

            this.startIndex = initialData.startIndex;//开始序号
            this.endIndex = initialData.endIndex;//结束序号
            this.iscomplete = false;//是否结束 初始化时为false
            this.isScrollError = false;

            this.totalRecordCount = initialData.dataCount || (Math.pow(2, 53) - 1);//一共有多少条
            this.pageCount = initialData.pageCount;//当前页面请求的条数

            this.listTemplateFunctionId = initialData.listTemplateFunctionId;//生成列表的函数id

            this.bottomLoading = initialData.bottomLoading;
        },

        /*接口调用后的处理函数*/
        scrollAdapter: function (data, isScroll) {
            var scope = this.self;
            /*正常请求回调*/
            if (data && data.body && data.body.obj) {
                this.totalRecordCount = data.body.obj.page.dataCount;//总条数
                this.startIndex = this.startIndex + this.pageCount; // 更新请求开始序号 add index if data fetched successfully
                isScroll ? this.addListAct(data) : this.listAct(data);//是在后面继续添加（第二次加载） 还是直接显示列表 （第一次列表显示）
            } else {
                /*请求失败 处理函数*/
                this.errorEvt(isScroll);
            };
        },

        /*第一次加载 data是一个response*/
        listAct: function (data) {
            var scope = this.self;
            scope.hideLoading();
            /*隐藏底部加载样式*/
            window.scrollTo(0, 0);
            this.listRender(data.body.obj.list);
        },

        /*非第一次加载  data是一个response*/
        addListAct: function (data) {
            var scope = this.self;
            this.hideBottomLoading();
            /*隐藏底部加载样式*/
            this.addListRender(data.body.obj.list);
            /*传递list数据*/
        },

        /*错误处理函数*/
        errorEvt: function (isscroll) {
            var scope = this.self;
            scope.hideLoading();
            this.hideBottomLoading();
            if (isscroll) {
                scope.showToast({
                    text: '加载失败，请稍后再试'
                });
                this.isScrollError = true;
                this.iscomplete = false;
                scope.endPull();
                //scope.closePull();
                scope.showErrorMsg();
                return false;
            }
            ;
            //this.iscomplete = true;
            scope.endPull();
            window.scrollTo(0, 0);
            //scope.showErrorMsg();
        },

        /*初次加载*/
        listRender: function (listData) {
            var scope = this.self;
            if (listData && listData.length > 0) {
                if (this.endIndex > this.totalRecordCount) {// return some but it is the end
                    this.iscomplete = true;
                    this.NO_MORE_DOM && this.NO_MORE_DOM.show();//没有更多
                    //this.listcontainer.append(NO_RESULT_DOM); //TBD
                    scope.endPull();
                    scope.closePull();
                } else {
                    this.iscomplete = false;
                    scope.endPull();
                    this.NO_MORE_DOM && this.NO_MORE_DOM.hide();
                }

            } else {//nothing return at all. It is a empty list.
                //noresult
                console.log('no results');
                this.iscomplete = true;
                //this.NO_MORE_DOM && this.NO_MORE_DOM.show();
                this.NO_RESULT_DOM && this.NO_RESULT_DOM.show();
                scope.endPull();//TBD
                scope.closePull();
                this.NO_MORE_DOM && this.NO_MORE_DOM.hide();
            }

            if (scope.$el.find('.kai_widget').length > 0) {
                scope.reparse(this.listTplFun, {list: listData}, {clear: true});
                //scope.$el.find(".order-img").length > 0 && scope.$el.find(".order-img").lazyload({effect : "fadeIn"});
            } else {
                scope.reparse({list: listData}, {
                    templateFunctionId: this.listTemplateFunctionId,
                    clear: true
                });//重新解析列表，并清空已经有的内容
            }
        },

        /*添加异步加载获得的新内容*/
        addListRender: function (listData) {
            var scope = this.self;
            var listContainer = null;
            var instantListContainer = '';//新增加的列表部分
            if (this.listContainer.is('ul') || this.listContainer.hasClass('list_container')) {
                listContainer = this.listContainer;
            } else {
                listContainer = this.listContainer.find('.list_container');
                console.log(listContainer);
            }

            if (_.isArray(listData) && listData.length > 0) {
                instantListContainer = scope.reparse({list: listData}, {
                    keepOld: true,
                    templateFunctionId: this.listTemplateFunctionId
                });
                listContainer.append(instantListContainer ? $(instantListContainer) : "").show();
                /*添加到当前已经有的列表底部*/
            }

            /*已经加载完*/
            if (this.endIndex > this.totalRecordCount || this.startIndex > this.totalRecordCount) {
                this.iscomplete = true;
                this.NO_MORE_DOM && this.NO_MORE_DOM.show();
                this.hideBottomLoading();
                scope.endPull();
                scope.closePull();
            } else {
                /*没有加载完*/
                this.iscomplete = false;
                scope.endPull();
                this.NO_MORE_DOM && this.NO_MORE_DOM.hide();
            }
        },

        /*显示底部的 加载中*/
        showBottomLoading: function () {
            console.log('show bottom loading');
            if (this.bottomLoading) {
                this.bottomLoading.show();
            }
        },

        /*隐藏底部的 加载中 样式*/
        hideBottomLoading: function () {
            console.log('hide bottom loading');
            if (this.bottomLoading) {
                this.bottomLoading.hide();
                this.bottomLoading = null;
            }
        },

        /*获得当前位置*/
        getScroll: function () {
            var _touch = event.targetTouches[0].clientY;
        }
    };

    viewHelper.listHelper = listHelper;

    function getChangeViewFunction(viewName) {
        return function () {
            Kai.forward(viewName);
        }
    }

    viewHelper.getChangeViewFunction = getChangeViewFunction;
    return viewHelper;
});
