define([
  'dojo/Deferred',
  "dojo/_base/array"
],
  function (
    Deferred,
    arrayUtil
  ) {
  "use strict";

  //查看页面状态
  var getViewStatesColumnsDefinition = function (argument) {
    return arrayUtil.map([
      { label: "页面URI", field: "id", width: '250'},//
      { label: "父页面id", field: "parent", width: '150' },
      { label: "节点名称", field: "nodeName", width: '60' },
      //{ label: "页面名称", field: "viewName", width: '10%' },
      //{ label: "是否是共用页面", field: "inaccessible", width: '20' },
      { label: "默认访问", field: "isDefault",width:'50'},
      { label: "转向页面", field: "redirectView",  width: '150'},
      { label: "页面状态", field: "state",  width: '50'},
      { label: "前端", field: "fe",  width: '25'},
      { label: "后台", field: "api",  width: '25'},
    ], function (column) {
      return column;
    });
  };

  //查看接口状态
  var getInterfaceStatesColumnsDefinition = function (argument) {
    return arrayUtil.map([
      /*
      url:opt.url,
      loadedLength:loadedLength,
      status:'加载中',
      requestTime:new Date()
      */
      { label: "请求地址",field: "id", width: '250'},//
      //{ label: "请求描述",field: "description", width: '250'},//
      //{ label: "报文长度", field: "loadedLength",  width: '50'},
      { label: "请求时间", field: "requestTime",  width: '50',formatter:function(requestTime){
        if(requestTime){
          return requestTime.toTimeString();
        }else{
          return '';
        }
      }},
      { label: "状态更新时间", field: "statusChangeTime",  width: '50',formatter:function(statusChangeTime){
        if(statusChangeTime){
          return statusChangeTime.toTimeString();
        }else{
          return '';
        }
      }},
      { label: "耗时(ms)", field: "costTime",  width: '50'},
      { label: "状态", field: "status",  width: '50'},
      { label: "错误码", field: "statusText",  width: '50'},
    ], function (column) {
      return column;
    });
  };

  /*used to fetch data*/
  var dataFetcher = {
    /**
    获得所有view的列表
    */
    getViewsStatesList:function(){
      var self = this;
      return Kai._store.fetch();
    },

    /**
    获得所有的
    */
    getInterfaceStatesList:function(){
      var self = this;
      return Kai._interfaceStatesStore.fetch();
    }
  };

  return {
    dataFetcher:dataFetcher,
    getColumnsDefinition:getViewStatesColumnsDefinition,
    getInterfaceColumnsDefinition:getInterfaceStatesColumnsDefinition
  };
});
